import { Card, CardGrid } from '@astrojs/starlight/components';

<CardGrid stagger>
	<Card title="Native property widgets" icon="pencil">
		Pick a property and get the exact widget Obsidian's Properties panel would
		give you: a date picker for dates, a checkbox for booleans, chips for
		lists. Types are respected end to end.
		[Edit properties](/guides/edit-properties/)
	</Card>
	<Card title="Auto Properties" icon="list-format">
		Define the values you use over and over - like a `status` list - and set
		them from a prompt instead of typing. Single or multi-select, with
		descriptions and learn-as-you-go values.
		[Set up Auto Properties](/guides/auto-properties/)
	</Card>
	<Card title="Bulk edit" icon="open-book">
		Right-click a folder or a selection of notes and set a property across all
		of them, with explicit skip, merge, or overwrite handling for notes that
		already have it.
		[Bulk edit metadata](/guides/bulk-edit/)
	</Card>
	<Card title="Automation" icon="rocket">
		Progress Properties keep task counts up to date on their own, and the
		Kanban helper updates linked notes when their card changes lane.
		[Automate task counts](/guides/progress-properties/)
	</Card>
	<Card title="Tags, edited where they live" icon="approve-check">
		Rename a body `#tag` in place or edit a frontmatter `tags:` list as a real
		list - MetaEdit handles both homes correctly.
		[Edit tags](/guides/edit-tags/)
	</Card>
	<Card title="A developer API" icon="setting">
		Update, create, and read metadata from Templater templates, Dataview
		buttons, and other plugins - including nested YAML paths and a
		metadata-change subscription.
		[Explore the API](/api/overview/)
	</Card>
</CardGrid>
